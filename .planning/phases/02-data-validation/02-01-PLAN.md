    ```javascript
    /**
     * Coerce a cell value to a number or null.
     * @param {any} val - Raw cell value from Sheets API
     * @returns {number|null}
     */
    function toNumberOrNull(val) {
      if (val === null || val === undefined || val === '') return null;
      const n = Number(val);
      return Number.isFinite(n) ? n : null;
    }
    ```
